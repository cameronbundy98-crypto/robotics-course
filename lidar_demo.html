LiDAR Range Measurement

Drag the polygon body to translate  ·  drag a yellow vertex to reshape  ·  pick a preset to reset the shape. Right panel: r(θ)r(\theta), with θ=0\theta{=}0 pointing up. 

function rayPolyDist(dx, dy, verts) {
  const n = verts.length;
  let tMin = null;
  for (let i = 0; i < n; i++) {
    const ax = verts[i].x,       ay = verts[i].y;
    const bx = verts[(i+1)%n].x, by = verts[(i+1)%n].y;
    const ex = bx - ax, ey = by - ay;
    const D  = dy * ex - dx * ey;
    if (Math.abs(D) < 1e-10) continue;
    const t = (ay * ex - ax * ey) / D;
    const u = (ay * dx - ax * dy) / D;
    if (t > 1e-4 && u >= -1e-9 && u <= 1 + 1e-9)
      if (tMin === null || t < tMin) tMin = t;
  }
  return tMin;
}

// ── Preset shapes ────────────────────────────────────────────────────────────
SHAPES = ({
  "Pentagon": d3.range(5).map(i => ({
    x: 1.8 * Math.sin(2*Math.PI*i/5),
    y: 5.5 + 1.8 * Math.cos(2*Math.PI*i/5)
  })),
  "Rectangle": [
    {x:-1.8,y:4.6},{x:1.8,y:4.6},{x:1.8,y:6.4},{x:-1.8,y:6.4}
  ],
  "Triangle": [
    {x:0.0,y:7.2},{x:2.3,y:3.9},{x:-2.3,y:3.9}
  ],
  "L-shape": [
    {x:-1.5,y:4.0},{x:0.5,y:4.0},{x:0.5,y:5.2},
    {x:-0.3,y:5.2},{x:-0.3,y:7.0},{x:-1.5,y:7.0}
  ],
  "Star": d3.range(10).map(i => ({
    x: (i%2===0?2.0:0.85)*Math.sin(2*Math.PI*i/10),
    y: 5.5+(i%2===0?2.0:0.85)*Math.cos(2*Math.PI*i/10)
  }))
})
viewof shapePreset = Object.assign(
  Inputs.radio(Object.keys(SHAPES), { value: "Pentagon" }),
  { style: "font-size:12px; margin:2px 0 6px 0;" }
)
// Both panels live in one cell → no OJS blocks inside column divs → no warnings.
// The right panel plot is rebuilt synchronously inside every redraw() call.
viewof objPos = {
  // ── Dimensions ──
  const LW = 385, LH = 335;           // left panel (scene)
  const RW = 330, RH = 300;           // right panel (plot)
  const pad = { t:10, r:10, b:28, l:28 };
  const iW = LW-pad.l-pad.r, iH = LH-pad.t-pad.b;

  const sx = d3.scaleLinear([-5,5],[0,iW]);
  const sy = d3.scaleLinear([0,10],[iH,0]);
  const maxRange = 9.5, N_RAYS = 180, N_SCAN = 400;

  // Mutable vertex array, seeded from the chosen preset
  const verts = SHAPES[shapePreset].map(v=>({...v}));

  // ── Left panel SVG ──
  const svg = d3.create("svg").attr("width",LW).attr("height",LH);
  const g   = svg.append("g").attr("transform",`translate(${pad.l},${pad.t})`);

  g.append("rect").attr("width",iW).attr("height",iH)
    .attr("fill","#11111e").attr("rx",5);

  g.append("g").selectAll(null).data(d3.range(-4,5,2)).enter().append("line")
    .attr("x1",d=>sx(d)).attr("x2",d=>sx(d)).attr("y1",0).attr("y2",iH)
    .attr("stroke","#2a2a40").attr("stroke-width",1);
  g.append("g").selectAll(null).data(d3.range(2,10,2)).enter().append("line")
    .attr("x1",0).attr("x2",iW).attr("y1",d=>sy(d)).attr("y2",d=>sy(d))
    .attr("stroke","#2a2a40").attr("stroke-width",1);

  [g.append("g").attr("transform",`translate(0,${iH})`).call(d3.axisBottom(sx).ticks(5).tickSize(3)),
   g.append("g").call(d3.axisLeft(sy).ticks(5).tickSize(3))
  ].forEach(ag=>{
    ag.selectAll("text").attr("fill","#888").attr("font-size","10px");
    ag.selectAll("line,path").attr("stroke","#555");
  });

  const rayGroup = g.append("g");

  const polyEl = g.append("polygon")
    .attr("fill","#2e6fb5").attr("fill-opacity",0.82)
    .attr("stroke","#7bb3f0").attr("stroke-width",1.5)
    .attr("stroke-linejoin","round").style("cursor","grab");

  const vertCircles = verts.map((_,i)=>
    g.append("circle").attr("r",7).attr("fill","#ffd166")
      .attr("stroke","#333").attr("stroke-width",1).style("cursor","grab")
  );
  const vertLabels = verts.map((_,i)=>
    g.append("text").attr("text-anchor","middle").attr("dominant-baseline","middle")
      .attr("fill","#222").attr("font-size","8px").attr("font-weight","bold")
      .attr("pointer-events","none").text(i+1)
  );

  g.append("circle").attr("cx",sx(0)).attr("cy",sy(0))
    .attr("r",7).attr("fill","#e63946").attr("stroke","#fff").attr("stroke-width",1.5);
  g.append("text").attr("x",sx(0)).attr("y",sy(0)-11)
    .attr("text-anchor","middle").attr("fill","#f88").attr("font-size","10px").text("LiDAR");

  // ── Right panel container ──
  const rightDiv = document.createElement("div");

  // ── Flex wrapper returned as the viewof value ──
  const container = document.createElement("div");
  container.style.cssText = "display:flex; gap:14px; align-items:flex-start;";
  container.appendChild(svg.node());
  container.appendChild(rightDiv);

  // ── Helpers ──
  function emit() {
    container.value = { verts: verts.map(v=>({...v})) };
    container.dispatchEvent(new CustomEvent("input"));
  }

  function buildPlot() {
    const scan = d3.range(N_SCAN).map(i=>{
      const theta = -Math.PI/2 + i*Math.PI/(N_SCAN-1);
      const dx=Math.sin(theta), dy=Math.cos(theta);
      const t = rayPolyDist(dx,dy,verts);
      const hit = t!=null && t<=maxRange;
      return {theta, dist:hit?t:maxRange, hit};
    });
    return Plot.plot({
      width:RW, height:RH,
      marginLeft:50, marginRight:15, marginBottom:45, marginTop:15,
      style:{fontSize:"12px"},
      x:{
        label:"angle θ (rad) →",
        domain:[-Math.PI/2, Math.PI/2],
        ticks:[-Math.PI/2,-Math.PI/4,0,Math.PI/4,Math.PI/2],
        tickFormat:d=>{
          const v=d/Math.PI;
          if(Math.abs(v+0.50)<0.01) return "-π/2";
          if(Math.abs(v-0.50)<0.01) return "π/2";
          if(Math.abs(v+0.25)<0.01) return "-π/4";
          if(Math.abs(v-0.25)<0.01) return "π/4";
          return "0";
        }
      },
      y:{label:"↑ range r(θ) (m)", domain:[0,10.2]},
      marks:[
        Plot.ruleY([maxRange],{stroke:"#bbb",strokeDasharray:"5,4",strokeWidth:1}),
        Plot.text([{theta:Math.PI/2*0.97,dist:maxRange}],{
          x:"theta",y:d=>d.dist+0.3,text:()=>"max range",
          fill:"#999",fontSize:10,textAnchor:"end"
        }),
        Plot.line(scan,{x:"theta",y:"dist",stroke:"#8899bb",strokeWidth:1.2}),
        Plot.dot(scan.filter(d=>d.hit),{
          x:"theta",y:"dist",fill:"#f4a261",r:2.2,opacity:0.9
        })
      ]
    });
  }

  function redraw() {
    // Rays
    const rays = d3.range(N_RAYS).map(i=>{
      const theta=-Math.PI/2+i*Math.PI/(N_RAYS-1);
      const dx=Math.sin(theta), dy=Math.cos(theta);
      const t=rayPolyDist(dx,dy,verts);
      const d=(t!=null&&t<=maxRange)?t:maxRange;
      return {x2:dx*d,y2:dy*d,hit:t!=null&&t<=maxRange};
    });
    const lines=rayGroup.selectAll("line").data(rays);
    lines.enter().append("line").merge(lines)
      .attr("x1",sx(0)).attr("y1",sy(0))
      .attr("x2",d=>sx(d.x2)).attr("y2",d=>sy(d.y2))
      .attr("stroke",d=>d.hit?"#f4a261":"#1e3d6e")
      .attr("stroke-width",d=>d.hit?1.0:0.7)
      .attr("opacity",d=>d.hit?0.85:0.55);
    lines.exit().remove();

    polyEl.attr("points",verts.map(v=>`${sx(v.x)},${sy(v.y)}`).join(" "));
    vertCircles.forEach((c,i)=>c.attr("cx",sx(verts[i].x)).attr("cy",sy(verts[i].y)));
    vertLabels.forEach((l,i)=>l.attr("x",sx(verts[i].x)).attr("y",sy(verts[i].y)));

    // Rebuild right-panel plot in place
    rightDiv.replaceChildren(buildPlot());
  }

  // ── Translate drag ──
  let prevPx, prevPy;
  polyEl.call(d3.drag()
    .on("start",function(event){
      [prevPx,prevPy]=d3.pointer(event,g.node());
      d3.select(this).style("cursor","grabbing");
    })
    .on("drag",function(event){
      const [px,py]=d3.pointer(event,g.node());
      const ddx=sx.invert(px)-sx.invert(prevPx);
      const ddy=sy.invert(py)-sy.invert(prevPy);
      verts.forEach(v=>{v.x+=ddx;v.y+=ddy;});
      [prevPx,prevPy]=[px,py];
      redraw(); emit();
    })
    .on("end",function(){d3.select(this).style("cursor","grab");})
  );

  // ── Vertex drag ──
  vertCircles.forEach((circle,i)=>{
    circle.call(d3.drag()
      .on("start",()=>circle.attr("r",9).style("cursor","grabbing"))
      .on("drag",function(event){
        const [px,py]=d3.pointer(event,g.node());
        verts[i].x=sx.invert(px);
        verts[i].y=sy.invert(py);
        redraw(); emit();
      })
      .on("end",()=>circle.attr("r",7).style("cursor","grab"))
    );
  });

  redraw();
  container.value = { verts: verts.map(v=>({...v})) };
  return container;
}
